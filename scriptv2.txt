function monitorDailyExpirations() {
  console.log("start");

  // --- 1. Variable Declarations & Configuration ---
  const EMAIL_COL = 6;
  const SHEET_NAME = "main";
  const SUBJECT_LINE = "ALERT: Crane GP/SC Expiration Notification -FMM Infra, TSL";
  const SEP_TEXT = `-------------------------------------------------\n`;
  
  let email_list = [];
  let final_email = "";
  let item_index = 1;

  // --- 2. Spreadsheet & Data Initialization ---
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(SHEET_NAME);
  const data = sheet.getDataRange().getValues();
  const headers = data[0];

  // --- 3. Email Template Constants ---
  const emailSubStart = `ALERT FOR GATEPASS/SAFETY-CHECK EXPIRATION:\n${SEP_TEXT}`;
  const emailTextEnd = `\n${SEP_TEXT}This message was sent to you from the Crane GatePass/SafetyCheck Monitoring System, FMM-Infrastructure, Tata Steel LTD.\nIf you want to update the GP/SP date or unsubscribe from this notification, please contact: alqama.hasnain@tatasteel.com or call +91-8050509173. \n\nThanks and Regards,\nAlqama H. (832659)\nTata Steel LTD.`;

  // --- 4. Date Initialization ---
  const today = new Date();
  today.setHours(0, 0, 0, 0);

  // --- 5. Extract Email Recipients ---
  for (let i = 1; i < data.length; i++) {
    if (data[i][EMAIL_COL]) {
      email_list.push(data[i][EMAIL_COL]);
    }
  }

  // --- 6. Main Processing Loop ---
  for (let i = 1; i < data.length; i++) {
    let row = data[i];
    let dateColD = row[3]; // GP VALID
    let dateColE = row[4]; // SAFETY CHECK

    let datesToCheck = [
      { date: dateColD, colName: headers[3] || "GP VALID" },
      { date: dateColE, colName: headers[4] || "SAFETY CHECK" }
    ];

    datesToCheck.forEach(item => {
      let targetDate = null;

      // Robust parsing to filter out "N/A" or malformed strings
      if (item.date instanceof Date) {
        targetDate = item.date;
      } else if (item.date !== "N/A") {
        let parsed = new Date(item.date);
        if (!isNaN(parsed.getTime())) {
          targetDate = parsed;
        }
      }

      // Check Expiration Logic
      if (targetDate) {
        targetDate.setHours(0, 0, 0, 0);

        let diffMs = targetDate.getTime() - today.getTime();
        let diffDays = Math.round(diffMs / (1000 * 60 * 60 * 24));

        let alertType = "";
        let alertSubject = "";
        let introMessage = "";

        if (diffDays <= 0) {
          alertType = "EXPIRED";
          alertSubject = `CRITICAL: ${item.colName} EXPIRED`;
        } else if (diffDays > 0 && diffDays <= 7) {
          alertType = "EXPIRING SOON";
          alertSubject = `WARNING: ${item.colName} EXPIRING IN ${diffDays} DAYS`;
        }

        // Build Record Details if expiration criteria met
        if (alertType !== "") {
          let rowDetails = headers.map((header, index) => {
            let cellValue = "";
            let headUpper = header.toString().toUpperCase();

            if (headUpper == "EMAIL" || !header || headUpper == "STATUS") {
              return "";
            } else {
              cellValue = row[index];
            }
            return `${header}: ${cellValue}`;
          }).filter(String).join('\n'); // filter(String) removes empty entries from the Email/Status skips

          let body = `${introMessage}\n=== RECORD DETAILS ===\n${rowDetails}`;

          // Append to final email string
          final_email += `\n${item_index}. ${alertSubject}\n${body}\n${SEP_TEXT}`;
          item_index++;
        }
      }
    });
  }

  // --- 7. Final Email Assembly & Dispatch ---
  if (final_email !== "") {
    final_email = emailSubStart + final_email + emailTextEnd;

    email_list.forEach(recipient => {
      // MailApp.sendEmail(recipient, SUBJECT_LINE, final_email);
      console.log("mail sent to", recipient);
    });

    console.log("FINAL OUTPUT:\n\n", final_email);
  } else {
    console.log("No expirations found today.");
  }

  console.log("end");
}
